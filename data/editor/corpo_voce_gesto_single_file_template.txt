            forma: {
                title: "FORMA",
                body: "Una scena sonora si riconosce quando il gruppo rende percepibili inizio, sviluppo, tensione e chiusura.",
                tags: ["inizio", "sviluppo", "tensione", "chiusura"]
            }
        };

        const scoreSymbols = [
            { type: "light", x: 84, y: 118 },
            { type: "intense", x: 156, y: 102 },
            { type: "hit", x: 228, y: 126 },
            { type: "gap", x: 314, y: 120 },
            { type: "growth", x: 408, y: 102 },
            { type: "density", x: 522, y: 124 },
            { type: "cut", x: 642, y: 110 }
        ];

        function populateParticles() {
            const fragment = document.createDocumentFragment();
            for (let index = 0; index < 18; index += 1) {
                const dot = document.createElement("span");
                dot.style.setProperty("--left", `${Math.random() * 100}%`);
                dot.style.setProperty("--top", `${18 + Math.random() * 78}%`);
                dot.style.setProperty("--size", `${1 + Math.random() * 3}px`);
                dot.style.setProperty("--delay", `${Math.random() * -14}s`);
                dot.style.setProperty("--duration", `${14 + Math.random() * 18}s`);
                fragment.appendChild(dot);
            }
            particleHost.appendChild(fragment);
        }

        function updateScrollProgress() {
            const scrollable = document.documentElement.scrollHeight - window.innerHeight;
            const ratio = scrollable > 0 ? window.scrollY / scrollable : 0;
            progressBar.style.width = `${Math.min(Math.max(ratio, 0), 1) * 100}%`;
        }

        function setCurrentPhase(index) {
            const safeIndex = Math.min(Math.max(index, 1), phaseSections.length);
            phaseCounter.textContent = `${safeIndex}/7`;
            phaseDots.forEach((dot, dotIndex) => {
                dot.classList.toggle("is-active", dotIndex === safeIndex - 1);
                dot.setAttribute("aria-current", dotIndex === safeIndex - 1 ? "step" : "false");
            });
        }

        function activateMapNode(key) {
            const item = mapContent[key];
            if (!item) {
                return;
            }
            mapNodes.forEach((node) => node.classList.toggle("is-active", node.dataset.node === key));
            mapReadoutTitle.textContent = item.title;
            mapReadoutBody.textContent = item.body;
            mapReadoutTags.innerHTML = item.tags.map((tag) => `<span>${tag}</span>`).join("");
        }

        mapNodes.forEach((node) => {
            const key = node.dataset.node;
            node.addEventListener("mouseenter", () => activateMapNode(key));
            node.addEventListener("focus", () => activateMapNode(key));
            node.addEventListener("click", () => activateMapNode(key));
        });

        const revealObserver = new IntersectionObserver((entries) => {
            entries.forEach((entry) => {
                if (entry.isIntersecting) {
                    entry.target.classList.add("is-visible");
                    if (entry.target === mapPanel) {
                        mapPanel.classList.add("is-visible");
                    }
                }
            });
        }, { threshold: 0.18 });

        document.querySelectorAll(".reveal").forEach((section) => revealObserver.observe(section));

        const phaseObserver = new IntersectionObserver((entries) => {
            const visible = entries
                .filter((entry) => entry.isIntersecting)
                .sort((first, second) => second.intersectionRatio - first.intersectionRatio);
            if (!visible.length) {
                return;
            }
            const phase = Number(visible[0].target.dataset.phase || 1);
            setCurrentPhase(phase);
        }, {
            threshold: [0.22, 0.42, 0.65],
            rootMargin: "-25% 0px -38% 0px"
        });

        phaseSections.forEach((section) => phaseObserver.observe(section));

        phaseDots.forEach((dot) => {
            dot.addEventListener("click", () => {
                const target = document.getElementById(dot.dataset.target);
                if (target) {
                    target.scrollIntoView({ behavior: body.classList.contains("reduce-motion") ? "auto" : "smooth", block: "start" });
                }
            });
        });

        motionToggle.addEventListener("click", () => {
            const pressed = motionToggle.getAttribute("aria-pressed") === "true";
            motionToggle.setAttribute("aria-pressed", String(!pressed));
            body.classList.toggle("reduce-motion", !pressed);
        });

        contrastToggle.addEventListener("click", () => {
            const pressed = contrastToggle.getAttribute("aria-pressed") === "true";
            contrastToggle.setAttribute("aria-pressed", String(!pressed));
            body.classList.toggle("high-contrast", !pressed);
        });

        textToggle.addEventListener("click", () => {
            const pressed = textToggle.getAttribute("aria-pressed") === "true";
            textToggle.setAttribute("aria-pressed", String(!pressed));
            body.classList.toggle("large-type", !pressed);
            resizeScoreCanvas();
        });

        let focusedSection = null;

        function clearFocusMode() {
            if (focusedSection) {
                focusedSection.classList.remove("is-focused");
            }
            focusedSection = null;
            body.classList.remove("is-focus-mode");
        }

        function toggleFocusMode(section) {
            if (focusedSection === section) {
                clearFocusMode();
                return;
            }
            clearFocusMode();
            focusedSection = section;
            focusedSection.classList.add("is-focused");
            body.classList.add("is-focus-mode");
        }

        phaseSections.forEach((section) => {
            section.addEventListener("dblclick", () => toggleFocusMode(section));
        });

        document.addEventListener("keydown", (event) => {
            if (event.key === "Escape") {
                clearFocusMode();
            }
        });

        let scoreAnimationFrame = null;
        let activeSymbolIndex = -1;

        function drawScore(activeIndex = -1) {
            const context = scoreCanvas.getContext("2d");
            const dpr = window.devicePixelRatio || 1;
            const rect = scoreCanvas.getBoundingClientRect();
            const width = rect.width;
            const height = rect.height;
            scoreCanvas.width = width * dpr;
            scoreCanvas.height = height * dpr;
            context.setTransform(dpr, 0, 0, dpr, 0, 0);
            context.clearRect(0, 0, width, height);

            context.strokeStyle = "rgba(255,255,255,0.16)";
            context.lineWidth = 1;
            context.beginPath();
            context.moveTo(40, height / 2);
            context.lineTo(width - 40, height / 2);
            context.stroke();

            scoreSymbols.forEach((symbol, index) => {
                const isActive = index === activeIndex;
                const alpha = isActive ? 1 : 0.68;
                const glow = isActive ? 18 : 0;
                context.save();
                context.globalAlpha = alpha;
                context.shadowBlur = glow;
                context.shadowColor = "rgba(255,255,255,0.85)";
                context.strokeStyle = "rgba(255,255,255,0.92)";
                context.fillStyle = "rgba(255,255,255,0.92)";
                context.lineWidth = 2;

                if (symbol.type === "light") {
                    context.beginPath();
                    context.arc(symbol.x, symbol.y, 4, 0, Math.PI * 2);
                    context.fill();
                } else if (symbol.type === "intense") {
                    context.beginPath();
                    context.arc(symbol.x, symbol.y, 10, 0, Math.PI * 2);
                    context.fill();
                } else if (symbol.type === "hit") {
                    context.beginPath();
                    context.moveTo(symbol.x, symbol.y - 28);
                    context.lineTo(symbol.x, symbol.y + 28);
                    context.stroke();
                } else if (symbol.type === "gap") {
                    context.globalAlpha = 0.35;
                    context.beginPath();
                    context.moveTo(symbol.x - 22, symbol.y);
                    context.lineTo(symbol.x + 22, symbol.y);
                    context.stroke();
                } else if (symbol.type === "growth") {
                    context.beginPath();
                    context.moveTo(symbol.x - 30, symbol.y + 14);
                    context.lineTo(symbol.x + 18, symbol.y - 14);
                    context.lineTo(symbol.x + 8, symbol.y - 14);
                    context.moveTo(symbol.x + 18, symbol.y - 14);
                    context.lineTo(symbol.x + 18, symbol.y - 4);
                    context.stroke();
                } else if (symbol.type === "density") {
                    [-18, -8, 2, 12, 22].forEach((offset, cloudIndex) => {
                        context.globalAlpha = 0.38 + cloudIndex * 0.11;
                        context.beginPath();
                        context.arc(symbol.x + offset, symbol.y + (cloudIndex % 2 === 0 ? -4 : 6), 4.4, 0, Math.PI * 2);
                        context.fill();
                    });
                } else if (symbol.type === "cut") {
                    context.beginPath();
                    context.moveTo(symbol.x - 16, symbol.y - 18);
                    context.lineTo(symbol.x + 16, symbol.y + 18);
                    context.moveTo(symbol.x + 16, symbol.y - 18);
                    context.lineTo(symbol.x - 16, symbol.y + 18);
                    context.stroke();
                }

                context.restore();
            });
        }

        function resizeScoreCanvas() {
            drawScore(activeSymbolIndex);
        }

        function stopScoreAnimation() {
            if (scoreAnimationFrame) {
                cancelAnimationFrame(scoreAnimationFrame);
                scoreAnimationFrame = null;
            }
        }

        function playScore() {
            stopScoreAnimation();
            if (body.classList.contains("reduce-motion")) {
                activeSymbolIndex = scoreSymbols.length - 1;
                drawScore(activeSymbolIndex);
                return;
            }

            const start = performance.now();
            const duration = 3600;

            function tick(now) {
                const elapsed = now - start;
                const progress = Math.min(elapsed / duration, 1);
                activeSymbolIndex = Math.min(scoreSymbols.length - 1, Math.floor(progress * scoreSymbols.length));
                drawScore(activeSymbolIndex);
                if (progress < 1) {
                    scoreAnimationFrame = requestAnimationFrame(tick);
                } else {
                    scoreAnimationFrame = null;
                }
            }

            scoreAnimationFrame = requestAnimationFrame(tick);
        }

        playScoreButton.addEventListener("click", playScore);
        window.addEventListener("resize", resizeScoreCanvas, { passive: true });
        window.addEventListener("scroll", updateScrollProgress, { passive: true });

        populateParticles();
        activateMapNode("gruppo");
        updateScrollProgress();
        resizeScoreCanvas();
        setCurrentPhase(1);
    </script>
</body>
</html>
